Q1- Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype?


 ANS:// Parent object constructor
        function Parent(name) {
        this.name = name;
        }
 
        // Method attached to the parent object
        Parent.prototype.greet = function() {
        console.log(`Hello, my name is ${this.name}`);
        };
 
        // Child object constructor
        function Child(name, age) {
        Parent.call(this, name);
        this.age = age;
        }
 
        // Set up the prototype chain between the parent and child objects
        Child.prototype = Object.create(Parent.prototype);
        Child.prototype.constructor = Child;
 
        // Create instances of the parent and child objects
        var parent = new Parent("John");
        var child = new Child("Sarah", 10);
 
       // Use the methods defined in the parent object
       parent.greet();  
  
       // Use the methods from the parent object's prototype in the child object
       child.greet();   
 
 OUTPUT: Hello, my name is John.
           Hello, my name is Sarah


Q2-Write code to explain prototype chaining?


ANS:// Parent object constructor
       function Parent(name) {
       this.name = name;
       }

       // Method attached to the parent object
       Parent.prototype.greet = function() {
       console.log(`Hello, my name is ${this.name}`);
       };

       // Child object constructor
       function Child(name, age) {
       Parent.call(this, name);
       this.age = age;
       }

       // Set up the prototype chain between the parent and child objects
       Child.prototype = Object.create(Parent.prototype);
       Child.prototype.constructor = Child;

       // Method specific to the child object
       Child.prototype.introduce = function() {
       console.log(`Hi, I'm ${this.name} and I'm ${this.age} years old`);
       };

       // Create an instance of the child object
       var child = new Child("Sarah", 10);

       // Use the methods from the parent object's prototype in the child object
       child.greet();    
       child.introduce();

OUTPUT:Hello, my name is Sarah
          Hi, I'm Sarah and I'm 10 years old.

      Array.prototype.sum = function() {

      };

Q4-Write a JavaScript function to retrieve all the names of object's own and inherited properties?


ANS:function getAllPropertyNames(obj) {
      var propertyNames = [];
  
      // Iterate over all properties in the object and its prototype chain
      for (var prop in obj) {
      propertyNames.push(prop);
      }
  
      return propertyNames;
      }
  
      // Example usage
      var obj = {
      name: "John",
      age: 30
      };
  
      var inheritedObj = Object.create(obj);
      inheritedObj.job = "Engineer";
  
      var propertyNames = getAllPropertyNames(inheritedObj);
  
      console.log(propertyNames);

OUTPUT:[ 'job', 'name', 'age' ]
